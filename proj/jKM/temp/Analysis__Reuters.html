                 * CREATE TABLE dumper
AS
SELECT substr(line, 0, instr(line, '{')-1) cluster_id
, str_to_map(substr(line, instr(line, '[')+1, length(line)- instr(line, '[')-2), ', ',':') points
FROM dumper_stage
;
                 * CREATE TABLE dumper_unpvt
AS
SELECT cluster_id, cast(article_id as int) article_id, cast(distance as float) distance
FROM dumper
LATERAL VIEW explode(points) exploded_table as article_id, distance
